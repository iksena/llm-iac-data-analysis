        }
    )

    template.has_resource_properties(
        "AWS::IAM::Role",
        {
            "AssumeRolePolicyDocument": {
                "Statement": [
                    {
                        "Principal": {
                            "Service": "events.amazonaws.com"
                        }
                    }
                ]
            }
        }
    )

    # Dead-letter SQS queue

        "AWS::SQS::Queue",
        {
            "RedrivePolicy": Match.object_like({

            })
        }
    )

    # Ensure exactly one of each required resource type
