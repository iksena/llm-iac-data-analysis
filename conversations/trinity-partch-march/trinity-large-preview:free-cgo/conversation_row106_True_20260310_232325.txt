            Events:
              - s3:ObjectCreated:*
  SNSTopic:
    Type: AWS::SNS::Topic
    Properties:
      TopicPolicy:
        Version: '2012-10-17'
        Statement:
          - Effect: Allow
            Principal:
              Service: s3.amazonaws.com
            Action: sns:Publish
            Resource: !Ref SNSTopic
            Condition:
              ArnLike:
                AWS:SourceArn: !Sub 'arn:${AWS::Partition}:s3:::${S3Bucket}'
              StringEquals:
                AWS:SourceAccount: !Ref 'AWS::AccountId'
