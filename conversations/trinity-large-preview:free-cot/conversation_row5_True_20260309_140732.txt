Description: DynamoDB table for ArtistId, Concert, and TicketSales tracking

    Type: AWS::DynamoDB::Table
    Properties:
      TableName: !Sub '${AWS::StackName}-ArtistConcertSales'
      BillingMode: PAY_PER_REQUEST
      AttributeDefinitions:
        - AttributeName: ArtistId
          AttributeType: S
        - AttributeName: ConcertId
          AttributeType: S
      KeySchema:
        - AttributeName: ArtistId
          KeyType: HASH
        - AttributeName: ConcertId
          KeyType: RANGE
      SSESpecification:
        Enabled: true
      Tags:
        - Key: Environment
          Value: !Ref AWS::StackName
