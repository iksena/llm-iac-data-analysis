Description: CloudWatch Dashboard with VPC Flow Logs Insights widgets for NAT Gateway analysis

Parameters:
  LogGroupName:
    Type: String
    Default: 'vpc-flow-logs'
    Description: Name of the VPC Flow Log group (e.g., 'vpc-flow-logs')

Resources:
  TopSourceIPsByOutboundTrafficQuery:
    Type: AWS::Logs::QueryDefinition
    Properties:
      Name: TopSourceIPsByOutboundTraffic
      LogGroupNames:
        - !Ref LogGroupName
      QueryString: |
        filter interfaceId like 'eni-' and action = 'accept' and trafficType = 'egress'
        | stats sum(bytes) as TotalBytes by srcAddr
        | sort TotalBytes desc
        | limit 20
