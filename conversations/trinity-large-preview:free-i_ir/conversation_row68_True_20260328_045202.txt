              - cloudwatch:PutMetricData
              - cloudwatch:GetMetricStatistics
              - cloudwatch:ListMetrics
              - ec2:DescribeTags
            Resource: '*'
      Description: Managed policy for CloudWatch integration
    Metadata:
      Effect: cloudwatch-policy

  SSMManagedPolicy:
    Type: AWS::IAM::ManagedPolicy
    Properties:
      PolicyDocument:
        Version: 2012-10-17
        Statement:
          - Effect: Allow
            Action:
              - ssm:UpdateInstanceInformation
              - ssmmessages:CreateControlChannel
              - ssmmessages:CreateDataChannel
              - ssmmessages:OpenControlChannel
              - ssmmessages:OpenDataChannel
            Resource: '*'
      Description: Managed policy for SSM integration
    Metadata:
      Effect: ssm-policy

  EC2InstanceProfile:
    Type: AWS::IAM::InstanceProfile
    Properties:
      Path: /
      Roles:
        - !Ref EC2Role
    Metadata:
      Effect: instance-profile

  # Attach managed policies to the role
  AttachCloudWatchPolicy:
    Type: AWS::IAM::Policy
    Properties:
      PolicyName: CloudWatchPolicyAttachment
      PolicyDocument:
        Version: 2012-10-17
        Statement:
          - Effect: Allow
            Action: '*'
            Resource: '*'
      Roles:
        - !Ref EC2Role
      ManagedPolicyArns:
        - !Ref CloudWatchPolicy
    DependsOn: CloudWatchPolicy

  AttachSSMManagedPolicy:
    Type: AWS::IAM::Policy
    Properties:
      PolicyName: SSMManagedPolicyAttachment
      PolicyDocument:
        Version: 2012-10-17
        Statement:
          - Effect: Allow
            Action: '*'
            Resource: '*'
      Roles:
        - !Ref EC2Role
      ManagedPolicyArns:
        - !Ref SSMManagedPolicy
    DependsOn: SSMManagedPolicy

Metadata:
  Invariants:
    - EC2 instances must have permissions to integrate with CloudWatch and SSM
    - Instance profile must be created for EC2 instances to assume the role
    - Managed policies should be attached to the IAM role for separation of concerns
