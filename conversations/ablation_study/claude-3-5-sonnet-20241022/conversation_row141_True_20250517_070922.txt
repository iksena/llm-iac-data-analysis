AWSTemplateFormatVersion: '2010-09-09'
Description: 'Automated image recognition pipeline using Step Functions, S3, and Rekognition'

Resources:
  ImageBucket:
    Type: 'AWS::S3::Bucket'
    DeletionPolicy: Delete
    Properties:
      NotificationConfiguration:
        EventBridgeConfiguration:
          EventBridgeEnabled: true

  ImageTopic:
    Type: 'AWS::SNS::Topic'
    Properties:
      DisplayName: 'Face Detection Alerts'

  TopicSubscription:
    Type: 'AWS::SNS::Subscription'
    Properties:
      Protocol: sms
      Endpoint: '+14071113333'
      TopicArn: !Ref ImageTopic

  StatesExecutionRole:
    Type: 'AWS::IAM::Role'
    Properties:
      AssumeRolePolicyDocument:
        Version: '2012-10-17'
        Statement:
          - Effect: Allow
            Principal:
              Service: states.amazonaws.com
            Action: 'sts:AssumeRole'
