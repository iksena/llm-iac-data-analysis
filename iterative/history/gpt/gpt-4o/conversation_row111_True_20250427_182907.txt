    NoEcho: true

Resources:
  DataDogApiKeySecret:
    Type: AWS::SecretsManager::Secret
    Properties:
      Name: DataDogApiKey
      SecretString: !Ref DataDogApiKey

  EventBridgeEventBus:
    Type: AWS::Events::EventBus
    Properties:
      Name: MyDataDogAppEventBus

  EventBridgeRole:
    Type: AWS::IAM::Role
