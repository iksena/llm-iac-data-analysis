  ECRRepository:
    Type: AWS::ECR::Repository

      RepositoryName: !Ref RepositoryName

    Type: AWS::ECR::RepositoryPolicy
    Properties:
      RepositoryName: !Ref ECRRepository
      PolicyText:
        Version: '2008-10-17'
        Statement:

            Effect: Allow
            Principal:
              AWS: !Sub 'arn:aws:iam::${AWS::AccountId}:root'
