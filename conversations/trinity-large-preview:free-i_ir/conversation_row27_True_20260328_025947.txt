The I-IR schema:

    {
      "id":      "<logical name, e.g. MyBucket>",

      "fields":  { "<prop>": "<value or ref>" },
      "effects": ["<tag>, e.g. public-read | encrypted | versioned"]
    }
  ],
  "edges": [
    { "type": "depends",  "from": "<id>", "to": "<id>" },
    { "type": "connects", "from": "<id>", "to": "<id>",
      "proto": "<tcp|udp|https>", "port": <number> }
  ],
  "invariants": ["<free-text constraint, e.g. encryption-at-rest required>"]
}
