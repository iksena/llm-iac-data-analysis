Message 3

[Generated Template Content]
AWSTemplateFormatVersion: '2010-09-09'
Description: Creates a private Amazon Route 53 Hosted Zone for internal DNS management within a VPC.

Parameters:
  VPCID:
    Type: AWS::EC2::VPC::Id
    Description: The ID of the VPC to associate with the private hosted zone.
  DomainName:
    Type: String
    Description: The domain name for the private hosted zone (e.g., example.com).
    Default: www.example.com

Resources:
  PrivateHostedZone:
    Type: AWS::Route53::HostedZone
    Properties:
      Name: !Ref DomainName
      VPCs:
        - VPCId: !Ref VPCID
          VPCRegion: !Ref 'AWS::Region'
      HostedZoneConfig:
        Comment: Private hosted zone for internal DNS management

Outputs:
  HostedZoneID:
    Description: The ID of the created private hosted zone.
    Value: !Ref PrivateHostedZone
    Export:
      Name: !Sub "${AWS::StackName}-HostedZoneID"...
