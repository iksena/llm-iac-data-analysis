  CloudFormation template to create an S3 bucket with a bucket policy that allows either
  enterprise-wide access based on an AWS Organization ID or restricts access to the current account if no Organization ID is provided.

Parameters:
  OrganizationId:
    Type: String
    Description: >
      The AWS Organization ID to allow access to.
      Leave empty for single-account access.
    Default: ''

Conditions:
  UseOrgAccess: !Not [ !Equals [ !Ref OrganizationId, '' ] ]

Resources:
  S3Bucket:
    Type: AWS::S3::Bucket
    Properties:
      AccessControl: Private

  S3BucketPolicy:
    Type: AWS::S3::BucketPolicy
    Properties:
      Bucket: !Ref S3Bucket
      PolicyDocument:
        Version: '2012-10-17'
        Statement:
          - !If 
            - UseOrgAccess
            - 
              Sid: OrgWideAccess
              Effect: Allow
              Principal: "*"
