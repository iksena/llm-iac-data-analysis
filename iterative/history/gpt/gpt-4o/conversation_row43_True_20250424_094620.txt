AWSTemplateFormatVersion: '2010-09-09'
Description: >
  CloudFormation template to create an S3 bucket with KMS encryption and lifecycle management.

Resources:
  S3Bucket:
    Type: 'AWS::S3::Bucket'
    Properties:
      BucketName: !Sub 'my-secure-bucket-${AWS::AccountId}'
      LifecycleConfiguration:
        Rules:
          - Id: "MoveToIA"
            Status: "Enabled"
            Transitions:
              - TransitionInDays: 30
                StorageClass: "STANDARD_IA"
          - Id: "ExpireObjects"
            Status: "Enabled"
            ExpirationInDays: 365
      BucketEncryption:
        ServerSideEncryptionConfiguration:
          - ServerSideEncryptionByDefault:
              SSEAlgorithm: aws:kms
              KMSMasterKeyID: !Ref KMSKey
      Tags:
        - Key: Environment
          Value: Production
        - Key: Department
          Value: Finance

  KMSKey:
    Type: 'AWS::KMS::Key'
    Properties:
      Description: 'KMS key for S3 bucket encryption'
      ...
