  KMSKey:
    Type: AWS::KMS::Key
    Properties:
      Description: "KMS key for centralized encryption management"
      KeyUsage: ENCRYPT_DECRYPT
      Origin: AWS_KMS
      EnableKeyRotation: true
      KeyPolicy:
        Version: "2012-10-17"
        Statement:
          - Sid: "EnableRootAndAdminAccess"
            Effect: Allow
            Principal:
              AWS: !Sub "arn:aws:iam::${AWS::AccountId}:root"
            Action: "kms:*"
            Resource: "*"

  KMSKeyAlias:
    Type: AWS::KMS::Alias
    Properties:
      AliasName: "alias/centralized-encryption"
      TargetKeyId: !Ref KMSKey
