    Type: AWS::Logs::QueryDefinition
    Properties:
      Name: ConnectionDurationByEndpoint
      LogGroupNames:
        - /aws/vpn/client
      QueryString: >-
        fields @timestamp, endpoint_id, duration_seconds
        | filter @message like "ClientDisconnected"
        | json @message auto
        | stats avg(duration_seconds) as avg_duration by endpoint_id
