AWSTemplateFormatVersion: '2010-09-09'
Description: CloudFormation template to create an S3 bucket with optional organization-wide access

Parameters:
  OrganizationId:
    Type: String
    Description: The ID of the AWS Organization (optional)
    Default: ''
  BucketName:
    Type: String
    Description: The name of the S3 bucket to create
    Default: ''

Resources:
  S3Bucket:
    Type: AWS::S3::Bucket
    Properties:
      BucketName: !Ref BucketName

  S3BucketPolicy:
    Type: AWS::S3::BucketPolicy
    Condition: HasOrganizationId
    Properties:
      Bucket: !Ref S3Bucket
      PolicyDocument:
        Version: '2012-10-17'
        Statement:
          - Effect: Allow
            Principal:
              AWS: !Sub 'arn:aws:iam::${AWS::AccountId}:root'
            Action: s3:*
            Resource:
              - !Sub 'arn:aws:s3:::${S3Bucket}'
              - !Sub 'arn:aws:s3:::${S3Bucket}/*'

Conditions:
  HasOrganizationId: !Not [!Equals [!Ref OrganizationId, '']]
