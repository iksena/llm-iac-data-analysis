AWSTemplateFormatVersion: '2010-09-09'
Description: CloudFormation template to create environment-specific DynamoDB tables for supporter product data.

Parameters:
  Stage:
    Type: String
    AllowedValues:
      - PROD
      - CODE
    Description: Environment stage (PROD for production, CODE for non-production)

Resources:
  SupporterProductDataTable:
    Type: AWS::DynamoDB::Table
    Properties:
      TableName: !Sub SupporterProductData-${Stage}
      BillingMode: !If [IsProd, PROVISIONED, PAY_PER_REQUEST]
      KeySchema:
        - AttributeName: id
          KeyType: HASH
      AttributeDefinitions:
        - AttributeName: id
          AttributeType: S
      ProvisionedThroughput:
        !If
          - IsProd
          -
            ReadCapacityUnits: 20
            WriteCapacityUnits: 5
          - !Ref AWS::NoValue
      SSESpecification:
        Enabled: true
      TimeToLiveSpecification:
        AttributeName: expiryDate
        Enabled: true
      Tags:
        - Key:...
