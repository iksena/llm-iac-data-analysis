Description: CloudFormation template to create environment-specific DynamoDB tables for supporter product data.

Parameters:
  Stage:
    Type: String
    Description: The environment stage (PROD or CODE)
    AllowedValues:
      - PROD
      - CODE
  TableNameSuffix:
    Type: String
    Description: Suffix for the DynamoDB table name to ensure uniqueness

Conditions:
  IsProd: !Equals [!Ref Stage, 'PROD']
  IsCode: !Equals [!Ref Stage, 'CODE']

Resources:
  SupporterProductDataTable:
    Type: AWS::DynamoDB::Table
    Properties:
      TableName: !Sub 'SupporterProductData-${Stage}-${TableNameSuffix}'
      AttributeDefinitions:
        - AttributeName: id
          AttributeType: S
      KeySchema:
        - AttributeName: id
          KeyType: HASH
      SSESpecification:
        SSEEnabled: true
      TimeToLiveSpecification:
        AttributeName: expiryDate
        Enabled: true
      BillingMode: !If
        - IsProd
        - PROVISIONED
