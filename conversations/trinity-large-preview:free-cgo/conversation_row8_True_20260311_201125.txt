    Type: AWS::Logs::QueryDefinition
    Properties:
      Name: TopSourceIPsOutbound
      LogGroupNames:
        - !Sub 'vpc-flow-logs-${AWS::AccountId}'
      QueryString: |
        filter interfaceId like 'eni-' and action = 'ACCEPT' and dstPort != 22
        | stats sum(bytes) as totalBytes by srcAddr
        | sort totalBytes desc
        | limit 10
