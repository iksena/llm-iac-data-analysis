                "ScaleOutCooldown": 60,

                },
                "DisableScaleIn": False
            }
        })

        "BillingMode": "PAY_PER_REQUEST",

            "Enabled": True
        },
        "TimeToLiveSpecification": {
            "AttributeName": "expiryDate",
            "Enabled": True
        }
    }
